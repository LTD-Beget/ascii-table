    }

    /**
     * @return bool
     */
    public function isControlChar() : bool
    {
        return $this->getValue() >= 0 && $this->getValue() <= 31;
    }

    /**
     * @return bool

    public function isPrintableChar() : bool
    {
        return $this->getValue() >= 32 && $this->getValue() <= 127;
